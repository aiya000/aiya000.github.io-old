Haskell製Lisp方言（+ inline-lisp for Haskell）

>>> print [parse|(1 2 3)|]
Cons (AtomInt 1) (Cons (AtomInt 2) (Cons (AtomInt 3) Nil))

>>> case AtomInt 123 of; [parse|123|] -> "good"
"good"

>>> fromSing (sing :: Sing [parse|10|])
AtomInt 10

>>> [parsePreprocess|'10|]
Quote (AtomInt 10)

>>> [zurae|(print 10)|]
10Nil

>>> [zurae|'10|]
AtomInt 10
